cmake_minimum_required(VERSION 3.5 FATAL_ERROR)

project(fdio-netopeer)

set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/CMakeModules")

set_property(GLOBAL PROPERTY USE_FOLDERS ON)

set(NOP_INCLUDE_DIRS ${PROJECT_BINARY_DIR}/lib)

## Enabled components
option(BUILD_NOP_KS "Build netopeer keystore" ON)
option(BUILD_NOP_SERVER "Build netopeer server" ON)
option(BUILD_NOP_CLI "Build netopeer cli" ON)

list(APPEND dir_options
  BUILD_NOP_KS
  BUILD_NOP_SERVER
  BUILD_NOP_CLI
)

set(BUILD_NOP_KS keystored)
set(BUILD_NOP_SERVER server)
set(BUILD_NOP_CLI cli)

## Add enabled components
foreach (opt ${dir_options})
  if (${opt})
    list(APPEND subdirs
      ${${opt}_DIR}
    )
  endif()
endforeach()

message(STATUS "Building the following subprojects: ${subdirs}")

foreach(dir ${subdirs})
  add_subdirectory(${dir})
endforeach()

include(Packager)
include(PackagerNetopeer)
make_packages()
